## Preprocessing the data

``` python
import pandas as pd

# Load the allinfo file of each individual
f1 = pd.read_csv('SC_Human_f1.UMI_filtered.ED3.allinfo.gz', index_col=0, header=None, sep='\t')
f2 = pd.read_csv('SC_Human_f2.UMI_filtered.ED3.allinfo.gz', index_col=0, header=None, sep='\t')
f3 = pd.read_csv('SC_Human_f3.UMI_filtered.ED3.allinfo.gz', index_col=0, header=None, sep='\t')
m1 = pd.read_csv('SC_Human_m1.UMI_filtered.ED3.allinfo.gz', index_col=0, header=None, sep='\t')
m2 = pd.read_csv('SC_Human_m2.UMI_filtered.ED3.allinfo.gz', index_col=0, header=None, sep='\t')
m3 = pd.read_csv('SC_Human_m3.UMI_filtered.ED3.allinfo.gz', index_col=0, header=None, sep='\t')

ct_mapping = {
    "ExciteDG": "ExcNeuron",
    "ExciteCA": "ExcNeuron",
    "InterneuronsCGE_VIP": "InhNeuron",
    "InterneuronsCGE_LAMP5": "InhNeuron",
    "InterneuronsMGE_PV": "InhNeuron",
    "InterneuronsMGE_SST": "InhNeuron",
    "AstrocytesProto": "Astro",
    "AstrocytesFibrous": "Astro",
    "MOLs": "Oligo",
    "MFOLs": "Oligo",
    "OPCs": "Oligo",
    "GranuleNB": "Granule",
    "Microglia": "Immune",
    "Macrophages": "Immune",
    "VascFibro": "Vascular",
    "VascEndo": "Vascular",
}

def process_allinfo(allinfo, sample_name):
    allinfo = allinfo[allinfo[10] > 0] #filter for spliced reads
    allinfo[2] = allinfo[2].map(ct_mapping) #relabel celltypes
    return allinfo

# Remove unspliced reads and relabel cell types to a lower resolution
dfs = {"f1": f1, "f2": f2, "f3": f3, "m1": m1, "m2": m2, "m3": m3}
dfs = {name: relabel_celltypes(df, name) for name, df in dfs.items()}
f1, f2, f3, m1, m2, m3 = (dfs[k] for k in ["f1","f2","f3","m1","m2","m3"])

# Create allinfo per celltype per sex
for ct in f1[2].unique()[:1]:

    f1_sub = f1[f1[2] == ct]
    f1_sub[2] = 'f1'
    
    f2_sub = f2[f2[2] == ct]
    f2_sub[2] = 'f2'
    
    f3_sub = f3[f3[2] == ct]
    f3_sub[2] = 'f3'
    
    m1_sub = m1[m1[2] == ct]
    m1_sub[2] = 'm1'
    
    m2_sub = m2[m2[2] == ct]
    m2_sub[2] = 'm2'

    m3_sub = m3[m3[2] == ct]
    m3_sub[2] = 'm3'

    f_sub = pd.concat((f1_sub, f2_sub, f3_sub), axis=0)
    m_sub = pd.concat((m1_sub, m2_sub, m3_sub), axis=0)

    f_path = f'allinfo_f_{ct}.gz'
    m_path = f'/allinfo_m_{ct}.gz'

    f_sub.to_csv(f_path, sep='\t', header=False, index=True)
    m_sub.to_csv(m_path, sep='\t', header=False, index=True)
    

```

## Step 2: Comparing exon inclusion between male and female 
scisorATAC can be downloaded [here](https://github.com/tilgnerlab/scisorATAC).


``` bash

celltypes=("Vascular" "ExcNeuron" "InhNeuron" "Astro" "Oligo" "Granule" "Immune")

base_allinfo_dir="human_hipp_data_analysis/"     # contains allinfo_{sex}_<celltype>.gz
chrom_file="human_hipp_data_analysis/all.chroms.1-22.X"
annotation_path="SC.extended_annotation.gtf.gz"
celltype_file="human_hipp_data_analysis/CellTypeFile_Ind"
output_base="human_hipp_data_analysis/exon_test/"
numThreads=32

# -----------------------------
# Loop over celltypes
# -----------------------------
for ((i=0; i<${#celltypes[@]}; i++)); do

    ct="${celltypes[i]}"

    outdir="${output_base}/${ct}"
    mkdir -p "$outdir"
    cd "$outdir"

    echo "=== Running ${ct} ==="
    echo "Output: $outdir"

    Rscript -e '
      library(scisorATAC)
      casesVcontrols(
        caseList = "'"${base_allinfo_dir}/allinfo_f_${ct}.gz"'",
        controlList = "'"${base_allinfo_dir}/allinfo_m_${ct}.gz"'",
        chrom_file = "'"${chrom_file}"'",
        numThreads = '"${numThreads}"',
        annotation_path = "'"${annotation_path}"'",
        ci_low = 0.05,
        ci_high = 0.95,
        min_reads = 10,
        zipping_function = "zcat",
        OL_fraction = 0.8,
        CellTypeFile = "'"${celltype_file}"'",
        OutputDir = "'"${outdir}"'"
      )
    '

done



```
